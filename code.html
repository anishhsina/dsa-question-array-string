1. Reverse an array in-place without using extra space

â–¼

Problem: Reverse an array without using additional space (in-place reversal).

Approach: Use two pointers - one at the beginning and one at the end. Swap elements and move pointers towards each other until they meet.

JavaScript Solution:

function reverseArray(arr) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left < right) {
        // Swap elements
        let temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;
        
        // Move pointers
        left++;
        right--;
    }
    
    return arr;
}
Time Complexity: O(n) | Space Complexity: O(1)

2. Find the maximum subarray sum (Kadane's Algorithm)

â–¼

Problem: Find the contiguous subarray with the maximum sum.

Approach: Kadane's algorithm keeps track of maximum sum ending at current position and global maximum.

JavaScript Solution:

function maxSubarraySum(arr) {
    if (arr.length === 0) return 0;
    
    let maxSoFar = arr[0];
    let maxEndingHere = arr[0];
    
    for (let i = 1; i < arr.length; i++) {
        // Either extend existing subarray or start new
        maxEndingHere = Math.max(arr[i], maxEndingHere + arr[i]);
        
        // Update global maximum
        maxSoFar = Math.max(maxSoFar, maxEndingHere);
    }
    
    return maxSoFar;
}
Time Complexity: O(n) | Space Complexity: O(1)

3. Find the duplicate number in an array

â–¼

Problem: Find the duplicate number in an array of n+1 integers where each integer is between 1 and n.

Approach: Floyd's Cycle Detection (Tortoise and Hare) - treat array as a linked list.

JavaScript Solution:

function findDuplicate(nums) {
    // Phase 1: Find intersection point in the cycle
    let tortoise = nums[0];
    let hare = nums[0];
    
    do {
        tortoise = nums[tortoise];
        hare = nums[nums[hare]];
    } while (tortoise !== hare);
    
    // Phase 2: Find entrance to the cycle
    tortoise = nums[0];
    while (tortoise !== hare) {
        tortoise = nums[tortoise];
        hare = nums[hare];
    }
    
    return hare;
}
Time Complexity: O(n) | Space Complexity: O(1)

4. Search in a rotated sorted array

â–¼

Problem: Search for a target value in a rotated sorted array.

Approach: Modified binary search - determine which half is sorted and check if target lies in that range.

JavaScript Solution:

function searchRotated(nums, target) {
    let left = 0;
    let right = nums.length - 1;
    
    while (left <= right) {
        let mid = Math.floor((left + right) / 2);
        
        if (nums[mid] === target) {
            return mid;
        }
        
        // Check which half is sorted
        if (nums[left] <= nums[mid]) {
            // Left half is sorted
            if (nums[left] <= target && target < nums[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else {
            // Right half is sorted
            if (nums[mid] < target && target <= nums[right]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    
    return -1; // Not found
}
Time Complexity: O(log n) | Space Complexity: O(1)

5. Merge two sorted arrays without extra space

â–¼

Problem: Merge two sorted arrays in-place without using extra space.

Approach: Start from the end of both arrays and place larger elements at the end of the first array.

JavaScript Solution:

function mergeSortedArrays(nums1, m, nums2, n) {
    let i = m - 1; // Last element in nums1
    let j = n - 1; // Last element in nums2
    let k = m + n - 1; // Last position in nums1
    
    // Merge from the end
    while (i >= 0 && j >= 0) {
        if (nums1[i] > nums2[j]) {
            nums1[k] = nums1[i];
            i--;
        } else {
            nums1[k] = nums2[j];
            j--;
        }
        k--;
    }
    
    // Copy remaining elements from nums2
    while (j >= 0) {
        nums1[k] = nums2[j];
        j--;
        k--;
    }
    
    return nums1;
}
Time Complexity: O(m + n) | Space Complexity: O(1)

6. Find the longest common prefix in a string array

â–¼

Problem: Find the longest common prefix string amongst an array of strings.

Approach: Compare characters vertically - check the same position across all strings.

JavaScript Solution:

function longestCommonPrefix(strs) {
    if (strs.length === 0) return "";
    
    // Start with first string as prefix
    let prefix = strs[0];
    
    for (let i = 1; i < strs.length; i++) {
        // Compare with each string and shorten prefix
        while (strs[i].indexOf(prefix) !== 0) {
            prefix = prefix.substring(0, prefix.length - 1);
            if (prefix === "") return "";
        }
    }
    
    return prefix;
}
Time Complexity: O(S) where S is sum of all characters | Space Complexity: O(1)

7. Check if a string is palindrome

â–¼

Problem: Check if a string reads the same forwards and backwards.

Approach: Use two pointers from both ends, comparing characters while moving towards center.

JavaScript Solution:

function isPalindrome(s) {
    // Clean string: remove non-alphanumeric and convert to lowercase
    const cleanStr = s.replace(/[^a-zA-Z0-9]/g, '').toLowerCase();
    
    let left = 0;
    let right = cleanStr.length - 1;
    
    while (left < right) {
        if (cleanStr[left] !== cleanStr[right]) {
            return false;
        }
        left++;
        right--;
    }
    
    return true;
}
Time Complexity: O(n) | Space Complexity: O(1)

8. Implement strstr() / indexOf()

â–¼

Problem: Find the first occurrence of a substring in a string.

Approach: Sliding window technique or KMP algorithm for optimal solution.

JavaScript Solution:

// Simple sliding window approach
function strStr(haystack, needle) {
    if (needle === "") return 0;
    
    for (let i = 0; i <= haystack.length - needle.length; i++) {
        let j = 0;
        
        // Check if substring matches
        while (j < needle.length && haystack[i + j] === needle[j]) {
            j++;
        }
        
        if (j === needle.length) {
            return i; // Found match
        }
    }
    
    return -1; // Not found
}
Time Complexity: O(n + m) with KMP, O(n*m) simple | Space Complexity: O(m) with KMP, O(1) simple

9. Rotate an array by k steps

â–¼

Problem: Rotate an array to the right by k steps.

Approach: Reverse the entire array, then reverse first k elements, then reverse remaining elements.

JavaScript Solution:

function rotateArray(nums, k) {
    k = k % nums.length; // Handle k > array length
    
    // Method 1: Three reversals
    reverse(nums, 0, nums.length - 1); // Reverse entire array
    reverse(nums, 0, k - 1);           // Reverse first k elements
    reverse(nums, k, nums.length - 1);     // Reverse remaining elements
    
    return nums;
}

function reverse(arr, start, end) {
    while (start < end) {
        let temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
}
Time Complexity: O(n) | Space Complexity: O(1)

10. Find missing number from 1â€¦n

Problem: Given an array containing n distinct numbers from 1 to n+1, find the missing number.

Approach: Use sum formula, XOR operation, or binary search depending on constraints.

JavaScript Solution:

// Method 1: Using sum formula
function findMissingNumber(nums) {
    const n = nums.length + 1;
    const expectedSum = (n * (n + 1)) / 2;
    const actualSum = nums.reduce((sum, num) => sum + num, 0);
    
    return expectedSum - actualSum;
}

// Method 2: Using XOR (handles overflow)
function findMissingNumberXOR(nums) {
    let missing = nums.length + 1; // Start with n+1
    
    for (let i = 0; i < nums.length; i++) {
        missing ^= nums[i] ^ (i + 1);
    }
    
    return missing;
}
Time Complexity: O(n) | Space Complexity: O(1)
